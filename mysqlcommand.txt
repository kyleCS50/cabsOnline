CREATE TABLE IF NOT EXISTS bookings (bookingID INT PRIMARY KEY AUTO_INCREMENT NOT NULL, 
refNum VARCHAR(50) NOT NULL, cname VARCHAR(255) NOT NULL, phone VARCHAR(12) NOT NULL, 
unumber VARCHAR(10), snumber VARCHAR(10) NOT NULL, stname VARCHAR(255) NOT NULL, sbname VARCHAR(255), 
dsbname VARCHAR(255), date DATE NOT NULL, time TIME NOT NULL, bookingDateTime DATETIME NOT NULL, 
assignStatus VARCHAR(255) NOT NULL);
